
/*
     Q15. Input a string from the user. Count occurrences (case insensitive) of
          each alphabet in the string.

          Sample output:

          Input: Welcome to SunBeam.
          Output:
          A : 1
          B : 1
          C : 1
          E : 3
          L : 1
          M : 2
          N : 1
          O : 2
          S : 1
          T : 1
          U : 1
          W : 1
*/


import java.util.Scanner;
import java.util.Arrays;

public class Q15_OccuranceCount 
{
    static void occurance(String s)
    {
        String copy = s;   //we created a copy so that original string remains intact.

        copy = copy.toUpperCase();

        char[] arr = copy.toCharArray();

        Arrays.sort(arr);

        copy = new String(arr);  //created a copy, so that original string doesnt get affected


        System.out.println("character occurance: \n");
        
        for(int hold = 0; hold < copy.length(); hold++ )
        {
            int count = 0;

            if(copy.charAt(hold) == ' ')
                {
                    continue;
                }

            else if(hold > 0 && copy.charAt(hold) == copy.charAt(hold-1))
                {
                    continue;
                }    
            
            else
            {    

                for(int move = 0; move < copy.length(); move++)
                {
                
                    if(copy.charAt(hold) == copy.charAt(move))
                    {
                        count++;
                    }
                }

            }

            System.out.printf("%c : %d\n", copy.charAt(hold), count);
        }

    }
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);

        System.out.println("Enter a string: ");

        String s = sc.nextLine();

        occurance(s);

        sc.close();

    }

    
}
